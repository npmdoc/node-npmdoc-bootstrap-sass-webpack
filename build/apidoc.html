function bootstrap-sass-webpack.bootstrap_sass_styles_loader (content)

  • description and source-code
    bootstrap_sass_styles_loader = function (content) {
  this.cacheable(true);
  var config = this.exec(content, this.resourcePath);
  var start =
      "@import          \"~bootstrap-sass/assets/stylesheets/bootstrap/variables\";\n"
    + "@icon-font-path: \"~bootstrap-sass/assets/fonts/bootstrap/\";\n"
    + "@import          \"./bootstrap-sass.config.scss\";\n";
  source = start + partials.filter(function (partial) {
    return config.styles[partial];
  }).map(function (partial) {
    return "@import \"~bootstrap-sass/assets/stylesheets/bootstrap/" + partial;
  }).join("\n");
  return source;
}

function bootstrap-sass-webpack.bootstrap_sass_scripts_loader.pitch (configPath)

  • description and source-code
    pitch = function (configPath) {
  this.cacheable(true);
  var config = require(configPath);
  return scripts.filter(function (script) {
    return config.scripts[script];
  }).map(function (script) {
    return "require(" + JSON.stringify("bootstrap-sass/assets/javascripts/bootstrap/" + script) + ");";
  }).join("\n");
}
